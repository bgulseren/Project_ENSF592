# Pyenv
```
# Get pyenv
curl https://pyenv.run | bash

# Get a list a avaliable python
pyenv install -l

# install a specific version
pyenv install 3.8.5

# Use Pyenv
eval "$(pyenv init -)"
pyenv versions
pyenv shell 3.8.5
python --version
virtualenv venv3.8.5
```
